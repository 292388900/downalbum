function randfloatnum()
	s = '0.'
	math.randomseed(os.time())
	for i = 1,16 do
		s = s .. math.random(0,9);
	end
	
	return s;
end

function GetOneAlbumPhotos(album)
	local perpagecount = 12
	local thispagecount = perpagecount
	local photocount = 0
	local photos = {}
	album.photos = {}
	ipage = 1
	
	ipages,_ =math.modf(album.total/perpagecount)
    rest = math.mod(album.total,perpagecount)
    ipages = ipages + (rest~=0 and 1 or 0)
	
	startno = 0
	for ipage = 1,ipages do
		
		if ipage==ipages then
			thispagecount = rest
		end
		
		url = string.format('http://%s/fcgi-bin/cgi_list_photo?g_tk=%s&callback=shine0_Callback&t=%s&mode=0&idcNum=%d&hostUin=%s&topicId=%s&noTopic=0&uin=%s&pageStart=%d&pageNum=%d&singleurl=1&notice=0&appid=4&inCharset=gbk&outCharset=gbk&source=qzone&plat=qzone&outstyle=json&format=jsonp&json_esc=1&callbackFun=shine0&_=%s',
		strPhotoRoute_S, g_tk,star.getcurrenttime(1),idcNum,hostUin,album.id,uin,startno,perpagecount,star.getcurrenttime(2))
		--msgbox(url)
		
		html = star.gethtml(url)
		if html==nil or html=='' then
			break
		end
		
		
		for name,origin,pre,url in string.gfind(html,[[name".-"(.-)".-origin_url".-"(.-)".-pre".-"(.-)".-"url".-"(.-)"]]) do
			if origin=='' then
				origin = url
			end
			origin = string.gsub(origin,[[\/]],'/')
			pre = string.gsub(pre,[[\/]],'/')
			photocount = photocount + 1
			photos[photocount] = {}
			photos[photocount].name = name
			photos[photocount].origin = origin
			photos[photocount].preview = origin
		end
		
		startno = startno + perpagecount
	end
	
	return photos
end
